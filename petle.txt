     <input type="number" name="liczba1" />
     <input type="number" name="liczba2" />
     <button> Oblicz NWD</button>
     <div class="wynik">Tu bedzie wynik obliczen</div>
     <script>
         //pobieramy elementy html do javascriptu
         const divW = document.querySelector('div.wynik');
         const btn = document.querySelector('button');

         //przypisujemy do przycisku btn zdarzenie click (klikanie mysza)
         btn.addEventListener('click', ()=>{
             //ten kod sie wykona w momencie klikniecia na button
             //pobranie wartosci pól input
             let l1 = document.querySelector('input[name="liczba1"]').value
             * 1;
             let l2 = document.querySelector('input[name="liczba2"]').value
             * 1;

             //obliczamy nwd wykorzystujac algorytm euklidesa
             while(l1 != l2){
                 if(l1 > l2){
                     l1 = l1 - l2;
                 }
                 else {
                     l2 = l2 - l1;
                 }
             }

             //wyswietlamy w divie informacje o NWD
             divW.textContent = 'NWD wynosi: ' + l1;
         });
     </script>
